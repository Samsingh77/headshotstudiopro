# Use Node.js 20 LTS Alpine for a lightweight, secure container environment
FROM node:20-alpine AS builder

WORKDIR /app

# Copy package files first to leverage Docker layer caching
COPY package*.json ./

# Install all dependencies (including devDependencies required for bundling)
RUN npm ci

# Copy the rest of the application files
COPY . .

# Build the frontend assets and compile the background Express server (server.ts -> dist/server.cjs)
RUN npm run build

# --- Production Runner Layer ---
FROM node:20-alpine AS runner

WORKDIR /app

ENV NODE_ENV=production
# Render/Railway/Cloud Run will automatically pass the PORT environment variable
ENV PORT=3000

# Copy the build configuration so we can prune/install production dependencies cleanly
COPY package*.json ./

# Install only production dependencies (excluding devDependencies to keep the image extremely small)
RUN npm ci --only=production

# Copy compiled build output (React static HTML/JS & server.cjs bundle) from the builder stage
COPY --from=builder /app/dist ./dist

# Expose port (Render/Railway route dynamically using PORT env var anyway)
EXPOSE 3000

# Start the Node.js Express server
CMD ["npm", "start"]
